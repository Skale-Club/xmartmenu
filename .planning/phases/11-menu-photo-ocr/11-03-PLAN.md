Three-step upload handler (D-01 flow — mandatory architecture):
```typescript
async function handleOcrUpload() {
  if (!ocrFile || !selectedMenuId) return
  setOcrLoading(true)
  setOcrStatus(null)
  try {
    // Step 1: get signed upload URL (filename for extension tracking, Pitfall 7)
    const tokenRes = await fetch(
      `/api/superadmin/tenants/${tenant.id}/ocr-upload-token?filename=${encodeURIComponent(ocrFile.name)}`
    )
    if (!tokenRes.ok) throw new Error('Failed to get upload token')
    const { uploadUrl, storagePath } = await tokenRes.json() as { uploadUrl: string; storagePath: string }

    // Step 2: PUT directly to Supabase Storage (bypasses Vercel 4.5 MB limit, D-01)
    const uploadRes = await fetch(uploadUrl, {
      method: 'PUT',
      body: ocrFile,
      headers: { 'Content-Type': ocrFile.type || 'image/jpeg' },
    })
    if (!uploadRes.ok) throw new Error('Storage upload failed')

    // Step 3: POST storage path to OCR process route (tiny payload, D-01)
    const processRes = await fetch(`/api/superadmin/tenants/${tenant.id}/ocr-process`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ storagePath, menuId: selectedMenuId }),
    })
    const data = await processRes.json() as {
      success?: boolean
      error?: string
      categoriesCreated?: number
      productsCreated?: number
    }
    if (!processRes.ok) throw new Error(data.error ?? 'OCR failed')
    setOcrStatus({
      type: 'success',
      message: `${data.categoriesCreated ?? 0} ${(data.categoriesCreated ?? 0) === 1 ? 'category' : 'categories'} and ${data.productsCreated ?? 0} ${(data.productsCreated ?? 0) === 1 ? 'product' : 'products'} extracted.`,
    })
  } catch (err) {
    setOcrStatus({
      type: 'error',
      message: err instanceof Error ? err.message : 'OCR failed — check API logs',
    })
  }
  setOcrLoading(false)
}
```
