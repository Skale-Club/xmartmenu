    ```typescript
    import { createServiceClient } from '@/lib/supabase/server'
    import { assertSuperadmin } from '@/lib/superadmin-auth'
    import { generateText, Output } from 'ai'
    import { openai } from '@ai-sdk/openai'
    import { revalidatePath } from 'next/cache'
    import { NextResponse } from 'next/server'
    import { OcrMenuSchema } from '@/lib/ai/schemas'

    // Node.js runtime required — @ai-sdk/openai uses Node APIs (D-05)
    export const runtime = 'nodejs'
    export const maxDuration = 60  // OCR is a single call; 60s matches text seed route (D-05)

    export async function POST(
      request: Request,
      { params }: { params: Promise<{ id: string }> }
    ) {
      const { id: tenantId } = await params

      // D-18: Auth guard — assertSuperadmin() first on every new route
      const supabase = await assertSuperadmin()
      if (!supabase) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

      const body = await request.json()
      const { storagePath, menuId } = body as { storagePath: string; menuId: string }

      if (!storagePath) return NextResponse.json({ error: 'storagePath is required' }, { status: 400 })
      if (!menuId) return NextResponse.json({ error: 'menuId is required' }, { status: 400 })

      const service = await createServiceClient()

      // Fetch tenant slug for revalidatePath (matches seed/route.ts pattern)
      const { data: tenant } = await service
        .from('tenants')
        .select('slug')
        .eq('id', tenantId)
        .single()

      // Validate menuId belongs to tenant
      const { data: menu } = await service
        .from('menus')
        .select('slug')
        .eq('id', menuId)
        .eq('tenant_id', tenantId)
        .single()

      if (!menu) return NextResponse.json({ error: 'Menu not found or does not belong to this tenant' }, { status: 404 })

      // ── Step 1: Download image from Storage ─────────────────────────────
      const { data: blobData, error: dlErr } = await service.storage
        .from('tenant-assets')
        .download(storagePath)

      if (dlErr || !blobData) {
        return NextResponse.json(
          { error: `Storage download failed: ${dlErr?.message ?? 'Unknown error'}` },
          { status: 500 }
        )
      }

      // Convert Blob → Buffer for AI SDK image part (confirmed Node.js pattern)
      const imageBuffer = Buffer.from(await blobData.arrayBuffer())

      // ── Step 2: GPT-4.1-mini vision extraction ───────────────────────────
      let ocrResult: { categories: Array<{ name: string; products: Array<{ name: string; price: number; description?: string | null }> }> }
      let totalTokens = 0

      try {
        const { output, usage } = await generateText({
          model: openai('gpt-4.1-mini'),
          output: Output.object({ schema: OcrMenuSchema }),
          messages: [
            {
              role: 'user',
              content: [
                {
                  type: 'text',
                  text: `Extract all menu categories and items from this restaurant menu photo.
    Return ONLY valid JSON matching the schema. Rules:
    - "name" for categories: the section header exactly as printed on the menu
    - "name" for products: the item name exactly as printed
    - "price": return as a NUMBER only (e.g. 32.00). Brazilian locale: replace comma with dot (32,50 → 32.50). Price ranges: use the lower value. If unreadable or missing: return 0.
    - "description": the printed description text exactly as shown, or null if not visible on the menu. DO NOT hallucinate descriptions.
    Include ALL categories and ALL items you can read. Do not skip any items.`,
                },
                {
                  type: 'image',
                  image: imageBuffer,
                },
              ],
            },
          ],
        })

        ocrResult = output
        totalTokens = usage?.totalTokens ?? 0
      } catch (err) {
        console.error('[ocr-process] GPT-4.1-mini error:', err)
        return NextResponse.json(
          { error: err instanceof Error ? err.message : 'OCR extraction failed — check server logs' },
          { status: 500 }
        )
      }

      // ── Step 3: Additive DB writes ───────────────────────────────────────
      // D-08: case-insensitive name match; never overwrite existing data
      let categoriesCreated = 0
      let productsCreated = 0

      try {
        // Fetch ALL existing categories for this tenant+menu upfront (avoids N+1)
        const { data: existingCats } = await service
          .from('categories')
          .select('name, id')
          .eq('tenant_id', tenantId)
          .eq('menu_id', menuId)

        const existingCatMap = new Map(
          (existingCats ?? []).map(c => [c.name.toLowerCase().trim(), c.id])
        )

        // Fetch max category position for ordering (D-09)
        const { data: maxCatPos } = await service
          .from('categories')
          .select('position')
          .eq('tenant_id', tenantId)
          .eq('menu_id', menuId)
          .order('position', { ascending: false })
          .limit(1)
          .single()

        let nextCatPosition = (maxCatPos?.position ?? -1) + 1

        // Fetch max product position for ordering (D-09)
        const { data: maxProdPos } = await service
          .from('products')
          .select('position')
          .eq('tenant_id', tenantId)
          .eq('menu_id', menuId)
          .order('position', { ascending: false })
          .limit(1)
          .single()

        let nextProdPosition = (maxProdPos?.position ?? -1) + 1

        for (const cat of ocrResult.categories) {
          const catKey = cat.name.toLowerCase().trim()
          let categoryId = existingCatMap.get(catKey)

          // Insert new category if not already exists (D-08: additive)
          if (!categoryId) {
            const { data: insertedCat, error: catErr } = await service
              .from('categories')
              .insert({
                tenant_id: tenantId,
                menu_id: menuId,
                name: cat.name,
                description: null,
                translations: {},
                position: nextCatPosition,
                is_active: true,
              })
              .select('id')
              .single()

            if (catErr) throw new Error(`Category insert failed: ${catErr.message}`)
            categoryId = insertedCat.id
            existingCatMap.set(catKey, categoryId)
            categoriesCreated++
            nextCatPosition++
          }

          if (!cat.products?.length) continue

          // Fetch existing products in this category upfront (D-08: avoid N+1)
          const { data: existingProds } = await service
            .from('products')
            .select('name')
            .eq('tenant_id', tenantId)
            .eq('category_id', categoryId)

          const existingProdNames = new Set(
            (existingProds ?? []).map(p => p.name.toLowerCase().trim())
          )

          // Insert new products only (D-08: skip duplicates by name)
          const prodsToInsert = cat.products
            .filter(p => !existingProdNames.has(p.name.toLowerCase().trim()))
            .map((p, i) => ({
              tenant_id: tenantId,
              menu_id: menuId,
              category_id: categoryId,
              name: p.name,
              description: p.description || null,
              price: p.price ?? 0,              // D-12: 0 for failed/missing price
              is_available: true,
              is_featured: false,
              tags: [],
              image_urls: [],
              position: nextProdPosition + i,
            }))

          if (prodsToInsert.length > 0) {
            const { data: insertedProds, error: prodErr } = await service
              .from('products')
              .insert(prodsToInsert)
              .select('id')

            if (prodErr) throw new Error(`Product insert failed: ${prodErr.message}`)
            productsCreated += insertedProds?.length ?? 0
            nextProdPosition += prodsToInsert.length
          }
        }
      } catch (err) {
        console.error('[ocr-process] DB write error:', err)
        return NextResponse.json(
          { error: err instanceof Error ? err.message : 'DB write failed — check server logs' },
          { status: 500 }
        )
      }

      // ── D-16: Log ai_usage (non-blocking) ────────────────────────────────
      try {
        const today = new Date().toISOString().slice(0, 10)
        await service.from('ai_usage').upsert({
          tenant_id: tenantId,
          feature_key: 'ocr_menu',
          date: today,
          call_count: 1,
          token_count: totalTokens,
        }, { onConflict: 'tenant_id,feature_key,date' })
      } catch (e) {
        console.error('[ai_usage] non-blocking log failed:', e)
      }

      // ── D-10: Revalidate ISR cache after all writes ────────────────────────
      if (tenant?.slug) {
        revalidatePath(`/${tenant.slug}`)
        if (menu?.slug) revalidatePath(`/${tenant.slug}/${menu.slug}`)
      }

      return NextResponse.json({
        success: true,
        categoriesCreated,
        productsCreated,
        tokensUsed: totalTokens,
      })
    }
    ```

    CRITICAL implementation notes:
    - Model string MUST be `openai('gpt-4.1-mini')` — NOT `gpt-4o-mini`, NOT `openai.chat('gpt-4.1-mini')`
    - Use `generateText` + `Output.object` (NOT deprecated `generateObject`) — per RESEARCH.md recommendation
    - Do NOT import or call `convertBufferToWebP` — OCR images stay as original JPEG/PNG (Pitfall 6)

    - `translations: {}` (empty object, not null) — matches seed/route.ts pattern for categories
    - The `categoryId` variable inside the for-loop must be `string` — TypeScript will infer it correctly from the insert + map logic
