```typescript
import { createServiceClient } from '@/lib/supabase/server'
import type { Tenant } from '@/types/database'

const customDomainCache = new Map<string, { tenant: Tenant; expires: number }>()
const CACHE_TTL_MS = 60_000 // 1 minute

export async function resolveTenantFromHost(host: string): Promise<Tenant | null> {
  // Strip port and lowercase
  const normalized = host.split(':')[0].toLowerCase()

  // Platform domain — not a custom domain
  const platformHost = process.env.NEXT_PUBLIC_APP_URL?.replace(/^https?:\/\//, '').split(':')[0] ?? 'xmartmenu.skale.club'
  if (normalized === platformHost) return null

  // Check cache
  const cached = customDomainCache.get(normalized)
  if (cached && cached.expires > Date.now()) return cached.tenant

  // DB lookup
  const supabase = await createServiceClient()
  const { data, error } = await supabase
    .from('tenants')
    .select('*')
    .eq('custom_domain', normalized)
    .eq('is_active', true)
    .single()

  if (error || !data) {
    customDomainCache.delete(normalized)
    return null
  }

  customDomainCache.set(normalized, { tenant: data, expires: Date.now() + CACHE_TTL_MS })
  return data
}
```

2. Update `src/middleware.ts` to check hostname before falling back to pathname:

Replace the current middleware content with:

```typescript
import { type NextRequest, NextResponse } from 'next/server'
import { updateSession } from './lib/supabase/middleware'
import { resolveTenantFromHost } from './lib/resolve-tenant-from-host'

const BLOCKED_TENANT_SLUGS = new Set([
  'pricing', 'features', 'about', 'faq', 'blog', 'help', 'support',
  'pt', 'en', 'legal', 'privacy', 'terms', 'contact', 'careers',
])

export async function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl
  const firstSegment = pathname.split('/')[1]
  const host = request.headers.get('host')?.split(':')[0].toLowerCase()

  // Block reserved slugs on platform domain only
  if (firstSegment && BLOCKED_TENANT_SLUGS.has(firstSegment)) {
    return NextResponse.json({ error: 'Not found' }, { status: 404 })
  }

  // Check if this is a custom domain request
  if (host) {
    const tenant = await resolveTenantFromHost(host)
    if (tenant) {
      // Rewrite: treat custom domain request as tenant slug request
      // e.g. sitedocliente.com/ -> xmartmenu.skale.club/joes-pizza
      const url = request.nextUrl.clone()
      url.pathname = `/${tenant.slug}${pathname === '/' ? '' : pathname}`
      return NextResponse.rewrite(url)
    }
  }

  return await updateSession(request)
}

export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)',
  ],
}
```

```typescript
export async function verifyDomainPointsToPlatform(domain: string): Promise<boolean> {
  const platformHost = process.env.NEXT_PUBLIC_APP_URL?.replace(/^https?:\/\//, '').split(':')[0] ?? 'xmartmenu.skale.club'
  try {
    const resolver = new Promise<boolean>((resolve) => {
      require('dns').lookup(domain, (err, address, family) => {
        if (err) { resolve(false); return }
        // Check if resolved address matches our platform
        require('dns').lookup(platformHost, (err2, platformAddr) => {
          if (err2 || !platformAddr) { resolve(false); return }
          resolve(address === platformAddr)
        })
      })
    })
    return await resolver
  } catch {
    return false
  }
}
```

2. Add a verification endpoint: `src/app/api/admin/tenants/[id]/verify-domain/route.ts`:

```typescript
import { NextResponse } from 'next/server'
import { verifyDomainPointsToPlatform } from '@/lib/verify-domain-dns'

export async function POST(
  request: Request,
  { params }: { params: Promise<{ id: string }> }
) {
  const { id } = await params
  const body = await request.json()
  const { custom_domain } = body

  if (!custom_domain) {
    return NextResponse.json({ error: 'custom_domain required' }, { status: 400 })
  }

  const verified = await verifyDomainPointsToPlatform(custom_domain)
  return NextResponse.json({ verified, domain: custom_domain })
}
```
