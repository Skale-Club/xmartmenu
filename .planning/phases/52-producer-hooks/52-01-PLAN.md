       ```typescript
       export interface EnqueueXphereOpts {
         eventId?: string   // Stripe event.id — forwarded to the note dedup
         tags?: string[]    // status/direction tags (e.g. ['status:active','upgrade'])
       }

       export async function enqueueXphereSync(
         tenantId: string,
         reason: SyncReason,
         opts?: EnqueueXphereOpts,
       ): Promise<void> {
         // Fail-open env gate (mirror rate-limit.ts): no client or no destination
         // URL → silent no-op. Onboarding / Stripe webhooks keep working.
         const url = resolveWorkerUrl()
         if (!client || !url) return
         try {
           await client.publishJSON({
             url,
             body: {
               tenantId,
               reason,
               ...(opts?.eventId ? { eventId: opts.eventId } : {}),
               ...(opts?.tags && opts.tags.length > 0 ? { tags: opts.tags } : {}),
             },
             retries: 5,
             deduplicationId: `xphere:${tenantId}:${reason}`,
           })
         } catch (err) {
           // NEVER rethrow — a CRM/QStash outage must not break onboarding or flip
           // a Stripe webhook 200→500. Log only (observe-and-swallow).
           console.error('xphere.enqueue_failed', {
             tenantId,
             reason,
             error: err instanceof Error ? err.message : String(err),
           })
         }
       }
       ```

    - `retries: 5`.
    - The publish target is `resolveWorkerUrl()` (XPHERE_WORKER_URL first, then NEXT_PUBLIC_APP_URL derived) — NEVER `req.url`, NEVER localhost-hardcoded.
    - The whole publish is inside one try/catch that swallows and returns void. There is NO throw path out of this function.
    - Only `eventId`/`tags` are spread into the body when present, so the worker's optional fields stay optional.

    Do NOT add a barrel `index.ts`. Do NOT call the Xphere client/network here — this file only PUBLISHES to QStash.
